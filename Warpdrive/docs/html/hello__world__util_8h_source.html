60void *mem_type_malloc(size_t length)
61{
62 void *ptr;
63
64 switch (test_mem_type) {
65 case UCS_MEMORY_TYPE_HOST:
66 ptr = malloc(length);
67 break;
68#ifdef HAVE_CUDA
69 case UCS_MEMORY_TYPE_CUDA:
70 CUDA_FUNC(cudaMalloc(&ptr, length));
71 break;
72 case UCS_MEMORY_TYPE_CUDA_MANAGED:
73 CUDA_FUNC(cudaMallocManaged(&ptr, length, cudaMemAttachGlobal));
74 break;
75#endif
76 default:
77 fprintf(stderr, "Unsupported memory type: %d\n", test_mem_type);
78 ptr = NULL;
79 break;

163ucs_memory_type_t parse_mem_type(const char *opt_arg)
164{
165 if (!strcmp(opt_arg, "host")) {
166 return UCS_MEMORY_TYPE_HOST;
167 } else if (!strcmp(opt_arg, "cuda") &&
168 check_mem_type_support(UCS_MEMORY_TYPE_CUDA)) {
169 return UCS_MEMORY_TYPE_CUDA;
170 } else if (!strcmp(opt_arg, "cuda-managed") &&
171 check_mem_type_support(UCS_MEMORY_TYPE_CUDA_MANAGED)) {
172 return UCS_MEMORY_TYPE_CUDA_MANAGED;
173 } else {
174 fprintf(stderr, "Unsupported memory type: \"%s\".\n", opt_arg);
175 }
176
177 return UCS_MEMORY_TYPE_LAST;
178}
